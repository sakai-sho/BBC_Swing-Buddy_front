# Azure App Service用の環境変数設定例
# これらの値をAzure App Serviceの設定で設定してください

# アプリケーション設定
NODE_ENV=production
PORT=8080

# Next.js設定
NEXT_TELEMETRY_DISABLED=1
NEXT_SHARP_PATH=/home/site/wwwroot/node_modules/sharp

# Node.js設定
NODE_OPTIONS=--max-old-space-size=2048

# Azure App Service設定
WEBSITE_NODE_DEFAULT_VERSION=20.19.3
WEBSITE_RUN_FROM_PACKAGE=1
WEBSITES_PORT=8080

# その他の必要な環境変数
# DATABASE_URL=your_database_url
# API_KEY=your_api_key

# 注意: Azure App Serviceの設定画面で「アプリケーション設定」セクションに
# 上記の環境変数を設定してください
# 特に重要なのは PORT=8080 と WEBSITES_PORT=8080 です
